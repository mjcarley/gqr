{\footnotesize\ttfamily gint gqr\+\_\+function\+\_\+roots (\begin{DoxyParamCaption}\item[{gdouble $\ast$}]{P,  }\item[{gdouble $\ast$}]{Q,  }\item[{gdouble $\ast$}]{R,  }\item[{gdouble}]{x0,  }\item[{gdouble}]{du0,  }\item[{gint}]{N,  }\item[{gdouble $\ast$}]{x,  }\item[{gdouble $\ast$}]{du }\end{DoxyParamCaption})}

Find the roots of the special function which solves the differential equation $p(x)u\textnormal{\textquotesingle}\textnormal{\textquotesingle}(x) + q(x)u\textnormal{\textquotesingle}(x) + r(x)u(x) = 0$, using the algorithm of Glaser, A., Liu, X. and Rokhlin, V., `A fast algorithm for the calculation of the roots of special functions', SIAM Journal on Scientific Computing, 29(4)\+:1420--1438, {\texttt{ http\+://dx.\+doi.\+org/10.\+1137/06067016X}}.


\begin{DoxyParams}{Parameters}
{\em P} & three element array containing the coefficients of $p(x)$, lowest order first; \\
\hline
{\em Q} & three element array containing the coefficients of $q(x)$, lowest order first; \\
\hline
{\em R} & three element array containing the coefficients of $r(x)$, lowest order first; \\
\hline
{\em x0} & initial root; \\
\hline
{\em du0} & derivative of function at {\itshape x0}, $u\textnormal{\textquotesingle}(x_{0})$; \\
\hline
{\em N} & number of roots and derivatives to be returned; \\
\hline
{\em x} & on return, the roots of $u(x)$; \\
\hline
{\em du} & on return, $u\textnormal{\textquotesingle}(x)$ at roots of $u(x)$.\\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
0 on success. 
\end{DoxyReturn}
